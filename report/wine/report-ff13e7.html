1075 * The preprocessor's lexographical grammar (approximately):

1077 * pp := {ws} # {ws} if {ws} {expr} {ws} \n
1078 * | {ws} # {ws} ifdef {ws} {id} {ws} \n
1079 * | {ws} # {ws} ifndef {ws} {id} {ws} \n
1080 * | {ws} # {ws} elif {ws} {expr} {ws} \n
1081 * | {ws} # {ws} else {ws} \n
1082 * | {ws} # {ws} endif {ws} \n
1083 * | {ws} # {ws} include {ws} < {anytext} > \n
1084 * | {ws} # {ws} include {ws} " {anytext} " \n
1085 * | {ws} # {ws} define {ws} {anytext} \n
1086 * | {ws} # {ws} define( {arglist} ) {ws} {expansion} \n
1087 * | {ws} # {ws} pragma {ws} {anytext} \n
1088 * | {ws} # {ws} ident {ws} {anytext} \n
1089 * | {ws} # {ws} error {ws} {anytext} \n
1090 * | {ws} # {ws} warning {ws} {anytext} \n
1091 * | {ws} # {ws} line {ws} " {anytext} " {number} \n
1092 * | {ws} # {ws} {number} " {anytext} " {number} [ {number} [{number}] ] \n
1093 * | {ws} # {ws} \n

1095 * ws := [ \t\r\f\v]*

1097 * expr := {expr} [+-*%^/|&] {expr}
1098 * | {expr} {logor|logand} {expr}
1099 * | [!~+-] {expr}
1100 * | {expr} ? {expr} : {expr}
1101 *
1102 * logor := ||
1103 *
1104 * logand := &&
1105 *
1106 * id := [a-zA-Z_][a-zA-Z0-9_]*
1107 *
1108 * anytext := [^\n]* (see note)
1109 *
1110 * arglist :=
1111 * | {id}
1112 * | {arglist} , {id}
1113 * | {arglist} , {id} ...
1114 *
1115 * expansion := {id}
1116 * | # {id}
1117 * | {anytext}
1118 * | {anytext} ## {anytext}
1119 *
1120 * number := [0-9]+
1121 *
1122 * Note: "anytext" is not always "[^\n]*". This is because the

1124 *

1126 * directives start at the beginning of the line, followed by a '#'

1155 * __LINE__ -> 123

1157 * __TIME__ -> "23:59:59"
